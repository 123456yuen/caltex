async function fetchPrice() { try { const response = await fetch('/api/fuel'); const htmlText = await response.text(); const parser = new DOMParser(); const doc = parser.parseFromString(htmlText, 'text/html'); // --- 偵錯用：將整頁文字印出來看看 --- console.log("網頁文字內容:", doc.body.innerText.substring(0, 1000)); // 只印前 1000 字 const rows = doc.querySelectorAll('tr'); for (let row of rows) { // 如果這行文字出現了，你可以看到程式正在讀取什麼 // console.log("檢查中:", row.innerText); if (row.innerText.includes('白金汽油')) { // 嘗試簡化關鍵字 console.log("找到關鍵字行:", row.innerText); const priceText = row.innerText.match(/([\d.]+)/); // 放寬 regex 測試 if (priceText) { document.getElementById('officialPrice').innerText = `HK$ ${priceText[1]}`; document.getElementById('officialPrice').dataset.price = priceText[1]; return; } } } throw new Error("找不到價格數據"); } catch (e) { document.getElementById('officialPrice').innerText = "讀取失敗"; console.error(e); } }
